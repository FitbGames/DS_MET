transform cm():
    on show:
        xpos 1.1
        easein 2.0 xpos 0.83
    on hide:
        easeout 2.0 xpos 1.1

transform button_slide(y = 60):
    on show:
        ypos 40
        parallel:
            easein 0.6 ypos y
        parallel:
            easein 0.6 alpha 1.0
    on hide:
        parallel:
            easeout 0.6 ypos 0
        parallel:
            easeout 0.6 alpha 0.0

screen mm_list():
    default c = 0
    zorder 9
    for k in mm_list.keys():
        if k != persistent.mm:
            $ c += 1
            vbox xalign 0.99 spacing 0 at button_slide(c * 43):
                if isinstance(mm_list[k], dict) and sum([val for val in mm_list[k].values() if val is not None]) > 1:
                    hbox:
                        spacing 0
                        xpos 5
                        $ v0, v1 = mm_list[k].keys()
                        button xpos 5 xsize 78 ysize 40 ypos 10 padding (0, 0, 0, 0):
                            background 'gui/main_menu/mm_bg_selector_left_idle.png'
                            hover_background 'gui/main_menu/mm_bg_selector_left_hover.png'
                            text v0 style "mm_button_text"
                            action [SetVariable('persistent.mm', v0), Hide('mm_list')]

                        button xpos -5 xsize 78 ysize 40 ypos 10 padding (0, 0, 0, 0):
                            background 'gui/main_menu/mm_bg_selector_right_idle.png'
                            hover_background 'gui/main_menu/mm_bg_selector_right_hover.png'
                            text v1 style "mm_button_text" xalign 0.6
                            action [SetVariable('persistent.mm', v1), Hide('mm_list')]

                else:
                    button xsize 145 ysize 40 ypos 10 padding (0, 0, 0, 0):
                        background 'gui/main_menu/mm_bg_selector_idle.png'
                        hover_background 'gui/main_menu/mm_bg_selector_hover.png'
                        text k style "mm_button_text"
                        action [SetVariable('persistent.mm', k), Hide('mm_list')]


screen language_confirm(lang, details = lang_details):
    modal True
    zorder 11
    add "gui/overlay/confirm.png"
    style_prefix 'confirm'
    frame:
        vbox xsize 500 spacing 40:
            hbox:
                xfill True
                text "{a}".format(a = details[lang]) text_align 0.5 size 20
            hbox:
                xfill True
                textbutton _("Cancel"):
                    action [Return(), Hide('language_confirm'), Hide('language_list')]
                textbutton _("Confirm") xalign 0.8:
                    action [Language(lang), Hide('language_confirm'), Hide('language_list')]

screen event_list():
    if config.developer == True:
        frame background None padding (0,0,0,0):
            add "#1e1e1fee"
            vpgrid xpos 10 xsize 1910:
                xfill True
                cols 3
                spacing 1
                draggable True
                mousewheel True
                transpose True

                for e in event_pool:
                    hbox:
                        text e.name size 18
                        if e.is_completed:
                            text ' {color=#4dc01a}completed{/color}' size 18
                        elif e.is_blocked:
                            text ' {color=#c0351a}blocked{/color}' size 18
                        elif e.is_available:
                            text ' {color=#cc6600}available{/color}' size 18
                        else:
                            text ' {color=#1ac09f}unknown{/color}' size 18
