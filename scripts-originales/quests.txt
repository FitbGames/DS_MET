init python:
    class Quest(object):
        def __init__(self, name = None, id = None, type = None, active = False, completed = False, objectives = [], current_step = None, resolution = None, tracked = False):
            self.name = name
            self.id = id
            self.type = type
            self.active = active
            self.completed = completed
            self.objectives = objectives
            self.current_step = current_step
            self.resolution = resolution
            self.tracked = tracked
            self.refresh_objectives()
            quests.append(self)


        def refresh_objectives(self):
            self.objectives = []
            for obj in quest_objectives:
                if self.id == obj.parent:
                    self.objectives.append(obj)

        def start(self, s = 1):
            if not self.active:
                self.active = True
                if self.current_step is None:
                    self.current_step = s
                renpy.notify(_('Quest "{}" started.'.format(self.name)))

        def advance(self):
            if self.current_step is None or self.current_step > len(self.objectives) + 1:
                pass
            else:
                self.current_step += 1
                renpy.notify(_('Quest "{}" updated.'.format(self.name)))

        def go_to_step(self, s):
            if self.current_step is None or s > len(self.objectives) + 1:
                pass
            else:
                self.current_step = s
                renpy.notify(_('Quest "{}" updated.'.format(self.name)))

        def complete(self):
            self.completed = True
            self.active = False
            renpy.notify(_('Quest "{}" completed.'.format(self.name)))

        def fail(self):
            self.completed = True
            self.active = False
            renpy.notify(_('Quest "{}" failed.'.format(self.name)))

        @property
        def is_tracked(self):
            return self.tracked

    class Objective(object):
        def __init__(self, name = None, parent = None, quest_step = None, description = '', condition = True, disabled = False, unavailable_message = ''):
            self.name = name
            self.parent = parent
            self.quest_step = quest_step
            self.description = description
            self.condition = condition
            self.disabled = disabled
            self.unavailable_message = unavailable_message
            self.add_to_list()

        @property
        def is_available(self):
            return eval(str(self.condition))

        @property
        def is_disabled(self):
            return eval(str(self.disabled))

        def add_to_list(self):
            quest_objectives.append(self)
