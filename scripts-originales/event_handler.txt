define event_pool = []

init python:
    class EventHandler(object):
        """
        """
        def __init__(self, label, condition, parent = None, location = None, blocked = None, tt = None, important = True, repeatable = False, completed = None, tracked = None, name = None):
            self.name = name
            self.parent = parent
            self.location = location
            self.label = label
            self.condition = condition
            self.blocked = blocked
            self.tt = tt
            self.important = important
            self.repeatable = repeatable
            self.completed = completed
            self.tracked = tracked
            self.validate()
            event_pool.append(self)

        def __str__(self):
            return str(self.name)

        def validate(self):
            if self.important and self.repeatable:
                raise AttributeError('Event cannot have the important and repeatable attributes at the same time.')

        @property
        def is_blocked(self):
            try:
                return eval(str(self.blocked))
            except NameError:
                raise
            except:
                raise SyntaxError(str(self.blocked))

        @property
        def is_available(self):
            try:
                return eval(str(self.condition))
            except NameError:
                raise
            except:
                raise SyntaxError('Malformed condition: {}'.format(self.condition))

        @property
        def is_completed(self):
            try:
                return eval(str(self.completed))
            except NameError:
                raise
            except:
                raise SyntaxError(str(self.completed))

        @property
        def is_active(self):
            try:
                return all([self.is_available and not self.is_completed and not self.is_blocked])
            except NameError:
                pass

        @property
        def is_tracked(self):
            return eval(str(self.tracked))

    def get_active_event(parent = None, location = None, events = event_pool):
        if isinstance(parent, list):
            valid_events = [x for x in events if ((x.parent in parent or x.location == location) and (x.parent is not None or x.location is not None) and x.is_active )]
        else:
            valid_events = [x for x in events if ((x.parent == parent or x.location == location) and (x.parent is not None or x.location is not None) and x.is_active)]
        if valid_events:
            for e in valid_events:
                if e.is_tracked:
                    return e
            for e in valid_events:
                if e.important:
                    return e    
            return valid_events.pop(0)
        else:
            return None    

    ##### Debug functions ####
    ### Show events that are still available to trigger
    def get_available_events(events = event_pool):
        return [e.name for e in events if e.important and e.is_available and not e.is_completed and not e.is_blocked]

    ### Show events that should be available still
    def get_incomplete_events(events = event_pool):
        return [e.name for e in events if e.important and not e.is_blocked and not e.is_completed]

    ### Show events that have unevaluated blocked conditions
    def check_blocked_conditions(events = event_pool):
        return [e.name for e in events if e.is_blocked not in [True, False, None]]
