    def time_format(t):
        if t is not None:
            m = str(int(t // 60))
            s = str(int((t % 60))).zfill(2)
            return ':'.join([m, s])
        else:
            return '0:00'

    class MusicTrack(object):
        def __init__(self, title, path, artist = None, category = [], environment = None, action = None, locked = False):
            self.title = title
            self.path = path
            self.artist = artist
            self.category = category
            self.environment = environment
            self.locked = locked
            music_room.add(self.path, always_unlocked=False)
            music_list.append(self)

        def __eq__(self, other):
            if isinstance(other, MusicTrack):
                return self.path == other.path


    class CustomMusicRoom(MusicRoom):
        def __init__(self, *args, **kwargs):
            super().__init__(*args, **kwargs)

        def is_unlocked(self, filename):
            for t in music_list:
                if t.path == filename:
                    return not t.locked
            return None

        def CycleLoop(self):
            if not self.loop:
                return [SelectedIf(False), self.SetLoop(True)]
            elif not self.single_track:
                return [SelectedIf(True), self.SetSingleTrack(True)]
            else:
                return [SelectedIf(True), self.SetSingleTrack(False),self.SetLoop(False)]

    music_room = CustomMusicRoom(fadein=2.0)

    class UpdatePlaying(Action):
        def __init__(self):
            self.playing = renpy.music.get_playing()

        def __call__(self):
            playing = renpy.music.get_playing()
            if self.playing != playing:
                self.playing = playing
            change_mr_bg(playing)

    def get_environment(track, l = music_list):
        if not renpy.music.is_playing():
            return 'stop'
        else:
            for t in l:
                if t.path == track:
                    return t.environment
        return None

    def get_artist(track, l = music_list):
        if not renpy.music.is_playing():
            return None
        else:
            for t in l:
                if t.path == track:
                    return t.artist
        return None

    def filter_playlist(playlist, filter):
        if not filter:
            return playlist
        else:
            return [track for track in playlist if all(f in track.category for f in filter)]

    def update_playlist_status(playlist, music_list = None):
        for track in music_list:
            if track in playlist:
                track.locked = False
            else:
                track.locked = True
        return [track for track in music_list if not track.locked]



    def change_mr_bg(track, env_list = music_room_envs):
        env = get_environment(track)
        if renpy.store.previous_env == env or env is None:
            pass
        else:
            renpy.store.previous_env = env
            mr_bg_intro = env_list.get(env, None).get('intro', None)
            mr_bg_loop = env_list.get(env, None).get('loop', None)
            mr_bg_sound = renpy.random.choice(env_list.get(env, None).get('sound', None))
            
            renpy.music.play(mr_bg_sound ,channel='sfx_1', loop = False)
            renpy.music.play(mr_bg_intro, channel='movie', loop = False)
            renpy.music.queue(mr_bg_loop, channel='movie', loop = True)
            renpy.restart_interaction()
