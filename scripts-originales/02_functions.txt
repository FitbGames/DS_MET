    ### Creator-Defined Statement for having random dialogue lines  
    ### https://patreon.renpy.org/three-creator-defined-statements.html#random
    def parse_random(l):
        l.require(":")
        l.expect_eol()
        blocks = [ ]
        ll = l.subblock_lexer()

        while ll.advance():
            with ll.catch_error():
                weight = 1.0
                if ll.keyword('weight'):
                    weight = float(ll.require(ll.float))
                blocks.append((weight, ll.renpy_statement()))
        return { "blocks" : blocks }

    def next_random(p):
        blocks = p["blocks"]
        # Pick a number between 0 and the total weight.
        total_weight = sum(i[0] for i in blocks)
        n = renpy.random.random() * total_weight
        for weight, block in blocks:
            if n <= weight:
                break
            else:
                n -= weight
        return block

    renpy.register_statement("random", parse=parse_random, next=next_random, predict_all=True, block=True)

    def parse_block(l):

        # Looks for a colon and the end of line.
        l.require(':')
        l.expect_eol()

        # Parses the block below this statement.
        block = l.subblock_lexer().renpy_block()

        return { "block" : block }

    # The next function returns the statement to execute next - in this case,
    # the first statement in the block.
    def next_block(p):
        return p["block"]

    renpy.register_statement("block", parse=parse_block, next=next_block, predict_all=True, block=True)

    ### Extend ConditionSwitch to support transitions (credit: Asceai @ lemmasoft)
    class TransitionConditionSwitch(renpy.Displayable):
        def __init__(self, transition, *args, **kwargs):
            super(TransitionConditionSwitch, self).__init__(**kwargs)
            self.transition = transition
            self.d = [(args[i], args[i+1]) for i in range(0, len(args), 2)]
            self.time_reset = True
            self.old_d = None
            self.current_d = None
            self.ta = None
            self.old_st = 0

        def render(self, width, height, st, at):
            if self.ta is None:
                self.per_interact()
            if self.time_reset:
                self.time_reset = False
                self.st = st
                self.at = at
                self.old_st = 0
            if st < self.old_st:
                self.st, self.at, st, at = 0, 0, 1000000.0, 1000000.0
            self.old_st = st
            return renpy.render(self.ta, width, height, st-self.st, at-self.at)

        def per_interact(self):
            for i, (name, d) in enumerate(self.d):
                if eval(name):
                    self.per_interact_change(name,d)
                    break

        def per_interact_change(self, name, change_to):
            if change_to is not self.current_d:
                self.time_reset = True
                self.old_d = self.current_d
                self.current_d = change_to
                if self.old_d is None:
                    self.old_d = self.current_d
                self.ta = anim.TransitionAnimation(self.old_d, 0.00, self.transition, self.current_d)
                renpy.redraw(self, 0)

        def visit(self):
            return [ self.ta ]

    def random_event(events, fallback, chance, re = True):
        """
        Picks an event at random from a pool of possible ones. Returns fallback event if below chance
        @param events: A dict of (event: condition) pairs.
        @param re: A boolean to make rollback preserve the outcome (renpy.random vs python random)
        """
        valid_events = [x for x in list(events.keys()) if eval(events[x]) is True]
        print(len(valid_events))
        if len(valid_events) > 0:
            if re:
                r = renpy.random.random()
                e = renpy.random.choice(valid_events)
            else:
                r = random.random()
                e = random.choice(valid_events)
            return fallback if r >= chance else e
        else:
            return fallback

    def random_image(events, coeff):
        """
        Picks an image at random from a pool of possible ones. Returns None if below threshold
        Chance is (valid / total) * coeff
        @param events: A dict of (image: condition) pairs.
        @param coeff: A Number that controls the probability of an event firing 
        """
        valid_events = [x for x in list(events.keys()) if eval(events[x]) is True]
        if renpy.random.random() <= (len(valid_events) / float(len(events.keys()))) * coeff:
            return random.choice(valid_events)
        else:
            return None

    from renpy.compat.pickle import loads

    def load(filename):
        """
        :doc: loadsave

        Loads the game state from the save slot `filename`. If the file is loaded
        successfully, this function never returns.
        """
        log_data, signature = renpy.loadsave.location.load(filename)
        persistent.last_fn = filename

        if not renpy.savetoken.check_load(log_data, signature):
            return
            
        roots, log = loads(log_data)
        log.unfreeze(roots, label="_after_load")

    renpy.load = load

    def restore_load(filename):
        log_data, signature = renpy.loadsave.location.load(filename)

        if not renpy.savetoken.check_load(log_data, signature):
            return
            
        roots, log = loads(log_data)

        with open('./save_trace.log', 'w') as output:
            new_ast = {}
            for k,v in renpy.game.script.namemap.items():
                if isinstance(v, renpy.ast.Say):
                    new_ast[(v.name[0], v.who, v.what)] = v.name

            for idx, item  in enumerate(log.log):
                print(f'___ Rollback log #{idx} ___', file=output)
                try:
                    print(f'Return Stack: {item.context.return_stack}', file=output)
                    print(f'Current: {item.context.current}, {renpy.game.script.lookup(item.context.current)}. Context OK', file=output)
                except Exception as e:
                    print(f'Current: {item.context.current} , failed script lookup with exception {e}', file=output)
                    try:
                        _last_statement = (item.context.current[0], item.stores['store']['_last_say_who'], item.stores['store']['_last_say_what'])
                        if _last_statement in new_ast:
                            print(f'Match found: Old key {item.context.current}, New key {new_ast[_last_statement]}. Remapping...', file=output)
                            item.context.current = new_ast[_last_statement]
                    except:
                        print(f'Failed to find match. Skipping...', file=output)
                        pass
                
                print(f'Stores: {item.stores}', file=output)
                print(f'AST ID: {item.identifier}\n', file=output)

        log.unfreeze(roots, label="_after_load")
