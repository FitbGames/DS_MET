init python:
    def get_btn_tooltip(e, default = None):
        try:
            if config.developer:
                return e.name
            else:
                if e.tt is not None:
                    return e.tt
                elif default is not None:
                    return default
                else:
                    return ''
        except:
            return ''

    def get_btn_foreground(e):
            if e.is_tracked:
                return 'gui/nav_menu/nav_tracked_overlay.png'
            elif e.important:
                return 'gui/nav_menu/nav_exclamation_overlay.png'
            else:
                return None

    def get_wm_foreground(e):
            if e.is_tracked:
                return 'gui/world_map/wm_tracked.png'
            elif e.important:
                return 'gui/world_map/wm_important.png'
            else:
                return None
