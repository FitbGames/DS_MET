init python:
    achievement_list = []

    class SteamAchievement(object):
        def __init__(self, name, title, description, condition):
            self.name = name
            self.title = title
            self.description = description
            self.condition = condition
            achievement_list.append(self)

        @property
        def is_unlocked(self):
            try:
                return eval(str(self.condition))
            except NameError:
                raise
            except:
                raise SyntaxError(str(self.condition))


    def register_achievements(list):
